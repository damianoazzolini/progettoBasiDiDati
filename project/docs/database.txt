UTENTE
CREATE TABLE utente (
	utenteID int AUTO_INCREMENT PRIMARY_KEY,
	nome varchar(255) NOT NULL,
	cognome varchar(255) NOT NULL,
	Email varchar(255) NOT NULL UNIQUE,
	Password varchar(255) NOT NULL,
	Citta varchar(255), ?
	DataNascita date
);

NOTIFICA
CREATE TABLE notifica (
	NotificaID int AUTO_INCREMENT PRIMARY_KEY,
	UserID varchar(255) FOREGIN_KEY REFERENCES utente(UserID),
	Tipo varchar(255),
	TipoID varchar(255),
	Letta bit DEFAULT 0
);

AMICIZIA
CREATE TABLE amicizia (
	UserID1 int FOREGIN KEY REFERENCES utente(UserID),
	UserID2 int FOREGIN_KEY REFERENCES utente(UserID),
	Data date,
	Ora time,
	Stato varchar(255) NOT_NULL DEFAULT 'sospesa'
);

COMMENTO
CRATE TABLE commento (
	CommentoID int AUTO_INCREMENT PRIMARY_KEY,
	UserID int FOREGIN_KEY REFERENCES utente(UserID),
	PostID int FOREGIN_KEY REFERENCES post(PostID),
	Contenuto text NOT_NULL,
	Data date,
	Ora time
);

POST
CREATE TABLE post (
	PostID int AUTO_INCREMENT PRIMARY_KEY,
	Contenuto text NOT_NULL,
	Data date,
	Ora time,
	UserID FOREGIN_KEY REFERENCES utente(UserID)
);

MEDIA
CREATE TABLE media (
	MediaID int AUTO_INCREMENT PRIMARY_KEY,
	PostID int FOREGIN_KEY REFERENCES post(PostID),
	Percorso text NOT_NULL
);

REAZIONE
CREATE TABLE rezione (
	ReazioneID int AUTO_INCREMENT PRIMARY_KEY,
	UserID int FOREGIN_KEY REFERENCES utente(UserID),
	PostID int FOREGIN_KEY REFERENCES post(PostID),
	Flag bit
);
